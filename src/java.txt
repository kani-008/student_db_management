 ComboBox<String> deptcom = new ComboBox<>();
        ObservableList<String> data = FXCollections.observableArrayList(
                "CSE", "ECE", "IT", "MECH", "CIV", "ATE");
        deptcom.setItems(data);
        deptcom.setEditable(true);
        deptcom.getEditor().textProperty().addListener((obs, oldVal, newVal) -> {
            ObservableList< String> filter = FXCollections.observableArrayList();
            for (String it : data) {
                if (it.toLowerCase().startsWith(newVal.toUpperCase()))
                    filter.add(it);
            }
            if(!deptcom.isshowing)
                deptcom.show();
            deptcom.setItems(filter);
            deptcom.setEditor().setText(newVal);
            deptcom.setEditor().positionCaret(newVal.length());
        });           






        FilteredList<String> filteredData = new FilteredList<>(data, s -> true);
        deptcom.setItems(filteredData);
        deptcom.setEditable(true);

// ðŸ” Filtering as you type
        deptcom.getEditor().textProperty().addListener((obs, oldVal, newVal) -> {
            String filter = newVal == null ? "" : newVal.toLowerCase();

            filteredData.setPredicate(item -> {
                if (filter.isEmpty()) {
                    return true;
                }
                return item.toLowerCase().startsWith(filter);
            });

            if (!filteredData.isEmpty()) {
                deptcom.show();
            } else {
                deptcom.hide();
            }
        });

        deptcom.getEditor().setOnKeyPressed(event -> {
            if (event.getCode() == KeyCode.ENTER) {
                String typed = deptcom.getEditor().getText();

                for (String item : data) {
                    if (item.equalsIgnoreCase(typed)) {
                        deptcom.setValue(item);
                        break;
                    }
                }
            }
        });
